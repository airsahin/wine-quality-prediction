# Use Python 3.10 
FROM python:3.10.13-slim-bullseye

# Set environment variables
ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    PIP_NO_CACHE_DIR=1

# Install system dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    curl \
    libgomp1 \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Copy requirements first for caching
COPY requirements.txt .
RUN pip install --upgrade pip==23.1.2 && \
    pip install --no-cache-dir -r requirements.txt

# Copy application (excluded files in .dockerignore)
COPY . .

# Create model directory with correct permissions
RUN mkdir -p model && \
    chmod -R 777 model && \
    chown -R 1000:1000 model  # Better for Railway's user context

EXPOSE 8501

# Production health check (now with timeout)
HEALTHCHECK --interval=30s --timeout=10s --start-period=10s --retries=3 \
    CMD curl -f http://localhost:8501/_stcore/health || exit 1

# Optimized Streamlit run command
CMD ["streamlit", "run", "app.py", \
    "--server.port=8501", \
    "--server.address=0.0.0.0", \
    "--server.headless=true", \
    "--server.enableCORS=false", \
    "--server.enableXsrfProtection=false", \
    "--browser.gatherUsageStats=false"]
